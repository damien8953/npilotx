async function refreshData() { document.getElementById('loader').style.display = 'block'; try { // 核心優化：只下載這一個 10KB 左右的小檔案 const response = await fetch(`all_atis.json?t=${new Date().getTime()}`); if (!response.ok) throw new Error("JSON not found"); const allData = await response.json(); for (const icao of allAirports) { const text = allData[icao]; if (!text) continue; const lines = text.split('\n'); const timeLine = lines[1] ? lines[1].replace(/Update:|更新時間:/, '').trim() : ""; const content = lines.slice(4).join('\n').trim(); // 更新 Download 時間 (這裡顯示的是 JSON 生成的時間) document.getElementById(`dl-${icao}`).innerText = `DL: ${getTimeAgo(timeLine)}`; // 更新 ATIS Zulu 與 相對時間 const atisInfo = getLatestZuluInfo(content); document.getElementById(`zulu-time-${icao}`).innerText = atisInfo.zulu; const ageEl = document.getElementById(`zulu-age-${icao}`); ageEl.innerText = atisInfo.diffText; ageEl.className = 'atis-age-span' + (atisInfo.isWarning ? ' warning' : ''); document.getElementById(`atis-${icao}`).innerText = content || "No current ATIS."; } } catch (e) { console.error("Fetch failed", e); // 如果 JSON 還沒生成，嘗試降級處理或顯示錯誤 } setTimeout(() => { document.getElementById('loader').style.display = 'none'; }, 500); }
